Programming Language Function Examples

Example 1: Python

Function Definition

  def greet(name):
      return f"Hello, [name]!"

Function Application

  message = greet("Alice")
  print(message)  # Output: Hello, Alice!

Example 2: JavaScript

Function Definition

  function multiply(a, b) [
      return a * b;
  ]

Function Application

  let result = multiply(4, 5);
  console.log(result); // Output: 20

Example 3: Java

Function Definition

  public static int addNumbers(int a, int b) [
      return a + b;
  ]

Function Application

  int result = addNumbers(3, 7);
  System.out.println(result); // Output: 10

Example 4: C++

Function Definition

  int subtract(int a, int b) [
      return a - b;
  ]

Function Application

  int result = subtract(10, 3);
  std::cout << result << std::endl; // Output: 7

Example 5: C#

Function Definition

  public static string SayHello(string name) [
      return $"Hello, [name]!";
  ]

Function Application

  string greeting = SayHello("John");
  Console.WriteLine(greeting); // Output: Hello, John!

Example 6: PHP

Function Definition

  function divide($a, $b) [
      return $a / $b;
  ]

Function Application

  $result = divide(20, 4);
  echo $result; // Output: 5

Example 7: Ruby

Function Definition

  def square(number)
    number * number
  end

Function Application

  result = square(6)
  puts result # Output: 36

Example 8: Swift

Function Definition

  func greet(name: String) -> String [return "Hello, \\(name)!"]

Function Application

  let message = greet(name: "Ella")
  print(message) // Output: Hello, Ella!

Example 9: Kotlin

Function Definition

  fun multiply(a: Int, b: Int): Int [
      return a * b
  ]

Function Application

  val result = multiply(5, 3)
  println(result) // Output: 15
